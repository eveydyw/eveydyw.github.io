[markup]
  [markup.goldmark]
    [markup.goldmark.extensions]
      [markup.goldmark.extensions.passthrough]
        enable = true
        [markup.goldmark.extensions.passthrough.delimiters]
          block = [['\[', '\]'], ['$$', '$$']]
          inline = [['\(', '\)']]
[params]
  math = true

inline & display

渲染数学公式时区分 inline 和 display 格式

  1. 更新 layouts/_default/_markup/render-passthrough.html 文件

    修改 render-passthrough.html 文件中的渲染逻辑，区分 inline 和 display 数学公式，并根据类型应用适当的 MathJax 渲染方式。

    {{ $opts := dict }}

{{ if eq .Type "block" }}
    {{ $opts = dict "displayMode" true }}
{{ end }}

{{ with try (transform.ToMath .Inner $opts) }}
    {{ with .Err }}
        {{ errorf "Failed to render MathJax: %q. See %s" . $.Position }}
    {{ else }}
        {{ $rendered := .Value }}

        <!-- 区分 inline 和 display -->
        {{ if $opts.displayMode }}
            <!-- 块级公式，MathJax 渲染 -->
            <div class="mathjax-display">
                {{ $rendered }}
            </div>
        {{ else }}
            <!-- 行内公式，MathJax 渲染 -->
            <span class="mathjax-inline">
                {{ $rendered }}
            </span>
        {{ end }}
    {{ end }}
{{ end }}

    // 行内数学公式样式
.mathjax-inline {
  display: inline-block;      // 保持在同一行
  font-size: 100%;            // 适中字体大小
  vertical-align: middle;     // 垂直居中
  padding: 0.2em 0.4em;       // 给公式添加内边距，让公式更显眼
  margin: 0;                  // 不增加外边距
  color: $primary;             // 设置颜色，可根据需求修改
  font-family: "Times New Roman", serif; // 字体样式，确保易读性
}

// 块级数学公式样式
.mathjax-display {
  display: block;             // 占据整行
  font-size: 100%;         
  margin: 20px 0;             // 上下外边距，增加间距
  padding: 0.5em;             // 给公式增加一些内边距
  text-align: center;         // 居中显示公式
  color: #1a73e8;             // 设置颜色，可根据需求修改
  font-family: "Times New Roman", serif; // 字体样式
  border: 1px solid #e0e0e0; // 为块级公式添加边框，增加区分度
  border-radius: 8px;         // 圆角效果
  background-color: #f4f6f9; // 背景色，使块级公式更加显眼
  box-shadow: 0 4px 8px rgba(0, 0, 0, 0.1); // 阴影效果，提升层次感
}
